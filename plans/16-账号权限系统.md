### 16.2 认证服务
**文件：`internal/app/service/auth_service.go`**
```go
type AuthService struct {
    userRepo     repository.UserRepository
    jwtManager   *auth.JWTManager
    passwordHasher *auth.PasswordHasher
}

func (s *AuthService) Login(ctx context.Context, username, password string) (string, string, error) {
    // 1. 查询用户
    user, err := s.userRepo.FindByUsername(ctx, username)
    if err != nil {
        return "", "", fmt.Errorf("invalid credentials")
    }
    
    // 2. 验证密码
    if !s.passwordHasher.Compare(password, user.Password) {
        return "", "", fmt.Errorf("invalid credentials")
    }
    
    // 3. 查询用户角色
    roles, err := s.userRepo.GetUserRoles(ctx, user.ID)
    if err != nil {
        return "", "", err
    }
    
    // 4. 生成Token
    token, err := s.jwtManager.GenerateToken(
        user.ID.String(),
        user.Username,
        user.DefaultWorkspaceID.String(),
        roles,
    )
    if err != nil {
        return "", "", err
    }
    
    // 5. 生成Refresh Token
    refreshToken, err := s.jwtManager.GenerateRefreshToken(user.ID.String())
    if err != nil {
        return "", "", err
    }
    
    return token, refreshToken, nil
}

func (s *AuthService) Register(ctx context.Context, username, email, password string) (*entity.User, error) {
    // 1. 验证用户名和邮箱唯一性
    if _, err := s.userRepo.FindByUsername(ctx, username); err == nil {
        return nil, fmt.Errorf("username already exists")
    }
    
    // 2. 创建用户
    hashedPassword := s.passwordHasher.Hash(password)
    
    user := &entity.User{
        ID:        uuid.New(),
        Username:  username,
        Email:     email,
        Password:  hashedPassword,
        Status:    "active",
        CreatedAt: time.Now(),
        UpdatedAt: time.Now(),
    }
    
    if err := s.userRepo.Create(ctx, user); err != nil {
        return nil, err
    }
    
    return user, nil
}
```

### 16.3 RBAC权限中间件
**文件：`internal/ports/http/middleware/rbac.go`**
```go
type RBACMiddleware struct {
    userRepo repository.UserRepository
}

func (m *RBACMiddleware) RequirePermission(permission string) gin.HandlerFunc {
    return func(c *gin.Context) {
        // 1. 从上下文获取用户信息
        claims, exists := c.Get("user")
        if !exists {
            c.AbortWithStatusJSON(401, gin.H{"error": "unauthorized"})
            return
        }
        
        userClaims := claims.(*auth.Claims)
        
        // 2. 验证权限
        hasPermission := m.checkPermission(
            c.Request.Context(),
            userClaims.UserID,
            userClaims.WorkspaceID,
            permission,
        )
        
        if !hasPermission {
            c.AbortWithStatusJSON(403, gin.H{"error": "forbidden"})
            return
        }
        
        c.Next()
    }
}

func (m *RBACMiddleware) checkPermission(ctx context.Context, userID, workspaceID, permission string) bool {
    // 权限格式：resource.action
    // 如：flows.create, flows.read, flows.update, flows.delete
    
    roles, err := m.userRepo.GetUserRolesInWorkspace(ctx, uuid.MustParse(userID), uuid.MustParse(workspaceID))
    if err != nil {
        return false
    }
    
    // 检查角色是否有权限
    for _, role := range roles {
        if m.roleHasPermission(role, permission) {
            return true
        }
    }
    
    return false
}
```

func (s *WorkspaceService) CreateWorkspace(ctx context.Context, name, description string, ownerID uuid.UUID) (*entity.Workspace, error) {
    workspace := &entity.Workspace{
        ID:          uuid.New(),
        Name:        name,
        Description: description,
        CreatedAt:   time.Now(),
        UpdatedAt:   time.Now(),
    }
    
    if err := s.workspaceRepo.Create(ctx, workspace); err != nil {
        return nil, err
    }
    
    // 添加创建者为workspace_owner
    member := &entity.WorkspaceMember{
        WorkspaceID: workspace.ID,
        UserID:      ownerID,
        Role:        "workspace_owner",
        JoinedAt:    time.Now(),
    }
    
    if err := s.workspaceMemberRepo.Create(ctx, member); err != nil {
        return nil, err
    }
    
    return workspace, nil
}

func (s *WorkspaceService) AddMember(ctx context.Context, workspaceID, userID uuid.UUID, role string) error {
    // 验证权限：只有workspace_owner可以添加成员
    
    member := &entity.WorkspaceMember{
        WorkspaceID: workspaceID,
        UserID:      userID,
        Role:        role,
        JoinedAt:    time.Now(),
    }
    
    return s.workspaceMemberRepo.Create(ctx, member)
}
