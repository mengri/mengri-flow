#### 10.2.1 创建资源

```go
func (s *ResourceService) CreateResource(ctx context.Context, req *dto.CreateResourceRequest) (*entity.Resource, error) {
    // 1. 验证资源类型插件是否存在
    plugin, err := s.pluginRegistry.GetResource(req.Type)
    if err != nil {
        return nil, fmt.Errorf("unsupported resource type: %s", req.Type)
    }
    
    // 2. 测试连接
    if err := plugin.TestConnection(ctx, req.Config); err != nil {
        return nil, fmt.Errorf("connection test failed: %w", err)
    }
    
    // 3. 创建资源实体
    resource := &entity.Resource{
        ID:          uuid.New(),
        Name:        req.Name,
        Type:        req.Type,
        Config:      req.Config,
        WorkspaceID: uuid.MustParse(req.WorkspaceID),
        Status:      "active",
        Description: req.Description,
        CreatedAt:   time.Now(),
        UpdatedAt:   time.Now(),
    }
    
    // 4. 保存到数据库
    if err := s.resourceRepo.Create(ctx, resource); err != nil {
        return nil, err
    }
    
    return resource, nil
}
```

#### 10.2.2 更新资源
```go
func (s *ResourceService) UpdateResource(ctx context.Context, id string, req *dto.UpdateResourceRequest) (*entity.Resource, error) {
    // 1. 查询资源
    resource, err := s.resourceRepo.FindByID(ctx, uuid.MustParse(id))
    if err != nil {
        return nil, err
    }
    
    // 2. 如果配置变更，测试新配置
    if req.Config != nil {
        plugin, err := s.pluginRegistry.GetResource(resource.Type)
        if err != nil {
            return nil, err
        }
        
        if err := plugin.TestConnection(ctx, req.Config); err != nil {
            return nil, fmt.Errorf("connection test failed: %w", err)
        }
        
        resource.Config = req.Config
    }
    
    // 3. 更新字段
    if req.Name != "" {
        resource.Name = req.Name
    }
    if req.Description != "" {
        resource.Description = req.Description
    }
    resource.UpdatedAt = time.Now()
    
    // 4. 保存
    if err := s.resourceRepo.Update(ctx, resource); err != nil {
        return nil, err
    }
    
    return resource, nil
}
```

#### 10.2.3 删除资源
```go
func (s *ResourceService) DeleteResource(ctx context.Context, id string) error {
    // 1. 检查是否有工具引用
    tools, err := s.toolRepo.ListByResource(ctx, uuid.MustParse(id))
    if err != nil {
        return err
    }
    
    if len(tools) > 0 {
        return fmt.Errorf("cannot delete resource with %d tools", len(tools))
    }
    
    // 2. 删除资源
    return s.resourceRepo.Delete(ctx, uuid.MustParse(id))
}
```

#### 10.2.6 批量导入工具
```go
func (s *ResourceService) ExtractTools(ctx context.Context, id string) ([]*entity.Tool, error) {
    // 1. 查询资源
    resource, err := s.resourceRepo.FindByID(ctx, uuid.MustParse(id))
    if err != nil {
        return nil, err
    }
    
    // 2. 获取插件
    plugin, err := s.pluginRegistry.GetResource(resource.Type)
    if err != nil {
        return nil, err
    }
    
    // 3. 调用插件的ExtractTools
    toolDefs, err := plugin.ExtractTools(ctx, resource.Config)
    if err != nil {
        return nil, err
    }
    
    // 4. 转换为Tool实体
    var tools []*entity.Tool
    for _, def := range toolDefs {
        tool := &entity.Tool{
            ID:           uuid.New(),
            Name:         def.Name,
            ResourceID:   resource.ID,
            Type:         def.Type,
            Method:       def.Method,
            Path:         def.Path,
            InputSchema:  def.InputSchema,
            OutputSchema: def.OutputSchema,
            Description:  def.Description,
            Status:       "draft",
            WorkspaceID:  resource.WorkspaceID,
            CreatedAt:    time.Now(),
            UpdatedAt:    time.Now(),
        }
        tools = append(tools, tool)
    }
    
    return tools, nil
}
```

### 10.4 配置脱敏
```go
func sanitizeConfig(config map[string]interface{}) map[string]interface{} {
    sensitiveKeys := []string{"password", "token", "apiKey", "secret"}
    
    sanitized := make(map[string]interface{})
    for k, v := range config {
        // 检查是否为敏感字段
        for _, sensitive := range sensitiveKeys {
            if contains(k, sensitive) {
                sanitized[k] = "******"
                break
            } else {
                sanitized[k] = v
            }
        }
    }
    
    return sanitized
}
```

```go
func RegisterResourceRoutes(router *gin.RouterGroup, service *service.ResourceService) {
    handler := NewResourceHandler(service)

    router.GET("/resources/:id", handler.GetResource)

    router.DELETE("/resources/:id", handler.DeleteResource)
```
