#### 11.2.1 创建工具
```go
func (s *ToolService) CreateTool(ctx context.Context, req *dto.CreateToolRequest) (*entity.Tool, error) {
    // 1. 验证资源存在且属于同一工作空间
    resource, err := s.resourceRepo.FindByID(ctx, uuid.MustParse(req.ResourceID))
    if err != nil {
        return nil, err
    }
    
    if resource.WorkspaceID != uuid.MustParse(req.WorkspaceID) {
        return nil, fmt.Errorf("resource not in workspace")
    }
    
    // 2. 创建工具实体
    tool := &entity.Tool{
        ID:           uuid.New(),
        Name:         req.Name,
        ResourceID:   uuid.MustParse(req.ResourceID),
        Type:         req.Type,
        Method:       req.Method,
        Path:         req.Path,
        InputSchema:  req.InputSchema,
        OutputSchema: req.OutputSchema,
        Description:  req.Description,
        Tags:         req.Tags,
        WorkspaceID:  uuid.MustParse(req.WorkspaceID),
        Status:       "draft",
        CurrentVersion: 0,
        CreatedAt:    time.Now(),
        UpdatedAt:    time.Now(),
    }
    
    // 3. 保存
    if err := s.toolRepo.Create(ctx, tool); err != nil {
        return nil, err
    }
    
    return tool, nil
}
```

#### 11.2.2 批量导入工具
```go
func (s *ToolService) ImportTools(ctx context.Context, req *dto.ImportToolsRequest) ([]*entity.Tool, error) {
    // 1. 查询资源
    resource, err := s.resourceRepo.FindByID(ctx, uuid.MustParse(req.ResourceID))
    if err != nil {
        return nil, err
    }
    
    // 2. 获取插件
    plugin, err := s.pluginRegistry.GetResource(resource.Type)
    if err != nil {
        return nil, err
    }
    
    // 3. 根据文件类型处理
    switch req.FileType {
    case "openapi":
        return s.importOpenAPI(ctx, resource, plugin, req.FileData)
    case "proto":
        return s.importProto(ctx, resource, plugin, req.FileData)
    case "sqlc":
        return s.importSQLc(ctx, resource, plugin, req.FileData)
    default:
        return nil, fmt.Errorf("unsupported file type: %s", req.FileType)
    }
}

func (s *ToolService) importOpenAPI(
    ctx context.Context,
    resource *entity.Resource,
    plugin plugin.ResourcePlugin,
    fileData []byte,
) ([]*entity.Tool, error) {
    // 1. 调用插件ExtractTools
    toolDefs, err := plugin.ExtractTools(ctx, resource.Config)
    if err != nil {
        return nil, err
    }
    
    // 2. 转换为Tool实体并保存
    var tools []*entity.Tool
    for _, def := range toolDefs {
        tool := &entity.Tool{
            ID:           uuid.New(),
            Name:         def.Name,
            ResourceID:   resource.ID,
            Type:         def.Type,
            Method:       def.Method,
            Path:         def.Path,
            InputSchema:  def.InputSchema,
            OutputSchema: def.OutputSchema,
            Description:  def.Description,
            WorkspaceID:  resource.WorkspaceID,
            Status:       "draft",
            CreatedAt:    time.Now(),
            UpdatedAt:    time.Now(),
        }
        
        if err := s.toolRepo.Create(ctx, tool); err != nil {
            // 记录错误但继续处理其他工具
            continue
        }
        
        tools = append(tools, tool)
    }
    
    return tools, nil
}
```

#### 11.2.4 发布工具
```go
func (s *ToolService) PublishTool(ctx context.Context, toolID string) (*entity.Tool, error) {
    // 1. 查询工具
    tool, err := s.toolRepo.FindByID(ctx, uuid.MustParse(toolID))
    if err != nil {
        return nil, err
    }
    
    // 2. 验证工具完整性
    if err := s.validateTool(ctx, tool); err != nil {
        return nil, fmt.Errorf("tool validation failed: %w", err)
    }
    
    // 3. 发布（状态变更 + 版本号+1）
    tool.Status = "published"
    tool.CurrentVersion++
    tool.UpdatedAt = time.Now()
    
    if err := s.toolRepo.Update(ctx, tool); err != nil {
        return nil, err
    }
    
    return tool, nil
}

func (s *ToolService) validateTool(ctx context.Context, tool *entity.Tool) error {
    // 验证必要字段
    if tool.Name == "" {
        return fmt.Errorf("name is required")
    }
    if tool.InputSchema == nil || tool.OutputSchema == nil {
        return fmt.Errorf("input/output schema is required")
    }
    
    return nil
}
```

#### 11.2.5 下线工具
```go
func (s *ToolService) DeprecateTool(ctx context.Context, toolID string) error {
    // 1. 检查是否有流程引用
    flows, err := s.flowRepo.ListByTool(ctx, uuid.MustParse(toolID))
    if err != nil {
        return err
    }
    
    if len(flows) > 0 {
        return fmt.Errorf("cannot deprecate tool referenced by %d flows", len(flows))
    }
    
    // 2. 更新状态
    tool, err := s.toolRepo.FindByID(ctx, uuid.MustParse(toolID))
    if err != nil {
        return err
    }
    
    tool.Status = "deprecated"
    tool.UpdatedAt = time.Now()
    
    return s.toolRepo.Update(ctx, tool)
}
```

#### 11.3.2 批量导入
```go
func (h *ToolHandler) ImportTools(c *gin.Context) {
    file, err := c.FormFile("file")
    if err != nil {
        response.BadRequest(c, "file is required")
        return
    }
    
    // 读取文件
    f, err := file.Open()
    if err != nil {
        response.Error(c, err)
        return
    }
    defer f.Close()
    
    fileData, err := io.ReadAll(f)
    if err != nil {
        response.Error(c, err)
        return
    }
    
    req := &dto.ImportToolsRequest{
        ResourceID: c.PostForm("resourceId"),
        FileType:   c.PostForm("fileType"),
        FileData:   fileData,
    }
    
    tools, err := h.service.ImportTools(c.Request.Context(), req)
    if err != nil {
        response.Error(c, err)
        return
    }
    
    response.Success(c, tools)
}
```
