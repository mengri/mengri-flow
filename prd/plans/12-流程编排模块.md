```go
func (s *FlowService) CreateFlow(ctx context.Context, req *dto.CreateFlowRequest) (*entity.Flow, error) {
    // 1. 验证画布数据完整性
    if err := s.validateCanvasData(req.CanvasData); err != nil {
        return nil, err
    }
    
    // 2. 验证所有工具节点引用的工具存在且已发布
    if err := s.validateToolNodes(req.CanvasData); err != nil {
        return nil, err
    }
    
    // 3. 创建流程实体
    flow := &entity.Flow{
        ID:           uuid.New(),
        Name:         req.Name,
        Description:  req.Description,
        CanvasData:   req.CanvasData,
        InputSchema:  req.InputSchema,
        OutputSchema: req.OutputSchema,
        WorkspaceID:  uuid.MustParse(req.WorkspaceID),
        Status:       "draft",
        CurrentVersion: 0,
        CreatedAt:    time.Now(),
        UpdatedAt:    time.Now(),
    }
    
    if err := s.flowRepo.Create(ctx, flow); err != nil {
        return nil, err
    }
    
    return flow, nil
}
```

```go
func (s *FlowService) validateCanvasData(canvasData map[string]interface{}) error {
    // 1. 验证nodes存在且是数组
    nodes, ok := canvasData["nodes"].([]interface{})
    if !ok || len(nodes) == 0 {
        return fmt.Errorf("canvas must have at least one node")
    }
    
    // 2. 验证edges存在
    edges, ok := canvasData["edges"].([]interface{})
    if !ok {
        return fmt.Errorf("canvas must have edges")
    }
    
    // 3. 验证必须有start和end节点
    hasStart := false
    hasEnd := false
    for _, n := range nodes {
        node := n.(map[string]interface{})
        nodeType := node["type"].(string)
        if nodeType == "start" {
            hasStart = true
        }
        if nodeType == "end" {
            hasEnd = true
        }
    }
    
    if !hasStart || !hasEnd {
        return fmt.Errorf("canvas must have start and end nodes")
    }
    
    return nil
}

func (s *FlowService) validateToolNodes(canvasData map[string]interface{}) error {
    nodes := canvasData["nodes"].([]interface{})
    
    for _, n := range nodes {
        node := n.(map[string]interface{})
        nodeType := node["type"].(string)
        
        if nodeType == "tool" {
            data := node["data"].(map[string]interface{})
            toolID := data["toolId"].(string)
            
            // 查询工具是否存在
            tool, err := s.toolRepo.FindByID(ctx, uuid.MustParse(toolID))
            if err != nil {
                return fmt.Errorf("tool %s not found", toolID)
            }
            
            // 验证工具已发布
            if tool.Status != "published" {
                return fmt.Errorf("tool %s is not published", toolID)
            }
        }
    }
    
    return nil
}
```

#### 12.2.3 更新流程
```go
func (s *FlowService) UpdateFlow(ctx context.Context, id string, req *dto.UpdateFlowRequest) (*entity.Flow, error) {
    flow, err := s.flowRepo.FindByID(ctx, uuid.MustParse(id))
    if err != nil {
        return nil, err
    }
    
    // 已发布的流程不允许编辑（需创建新版本）
    if flow.Status == "published" {
        return nil, fmt.Errorf("published flow cannot be edited, create new version instead")
    }
    
    // 更新字段
    if req.Name != "" {
        flow.Name = req.Name
    }
    if req.Description != "" {
        flow.Description = req.Description
    }
    if req.CanvasData != nil {
        if err := s.validateCanvasData(req.CanvasData); err != nil {
            return nil, err
        }
        flow.CanvasData = req.CanvasData
    }
    if req.InputSchema != nil {
        flow.InputSchema = req.InputSchema
    }
    if req.OutputSchema != nil {
        flow.OutputSchema = req.OutputSchema
    }
    
    flow.UpdatedAt = time.Now()
    
    if err := s.flowRepo.Update(ctx, flow); err != nil {
        return nil, err
    }
    
    return flow, nil
}
```

```go
func (s *FlowService) TestFlow(ctx context.Context, req *dto.TestFlowRequest) (*dto.TestFlowResponse, error) {
    // 1. 查询流程
    flow, err := s.flowRepo.FindByID(ctx, uuid.MustParse(req.FlowID))
    if err != nil {
        return nil, err
    }
    
    // 2. 创建运行记录
    run := &entity.Run{
        ID:          uuid.New(),
        FlowID:      flow.ID,
        FlowVersion: flow.CurrentVersion,
        Status:      "running",
        InputData:   req.Input,
        StartedAt:   time.Now(),
    }
    
    if err := s.runRepo.Create(ctx, run); err != nil {
        return nil, err
    }
    
    // 3. 执行流程（同步执行）
    executor := NewFlowExecutor(s.pluginRegistry, s.resourceRepo, s.toolRepo, s.nodeLogRepo)
    
    output, err := executor.Execute(ctx, flow, req.Input, run.ID)
    
    // 4. 更新运行记录
    run.Status = "success"
    if err != nil {
        run.Status = "failed"
        run.ErrorMessage = err.Error()
    }
    run.OutputData = output
    run.FinishedAt = time.Now()
    run.DurationMs = int(time.Since(run.StartedAt).Milliseconds())
    
    if err := s.runRepo.Update(ctx, run); err != nil {
        return nil, err
    }
    
    // 5. 查询节点日志
    nodeLogs, err := s.nodeLogRepo.ListByRun(ctx, run.ID)
    
    return &dto.TestFlowResponse{
        RunID:    run.ID.String(),
        Status:   run.Status,
        Output:   output,
        NodeLogs: convertNodeLogs(nodeLogs),
    }, err
}
```

#### 12.2.5 流程执行器
**文件：`internal/domain/flow/executor.go`**

```go
type FlowExecutor struct {
    pluginRegistry *plugin.Registry
    resourceRepo   repository.ResourceRepository
    toolRepo       repository.ToolRepository
    nodeLogRepo    repository.RunNodeLogRepository
}

func (e *FlowExecutor) Execute(
    ctx context.Context,
    flow *entity.Flow,
    input map[string]interface{},
    runID uuid.UUID,
) (map[string]interface{}, error) {
    // 1. 解析画布数据，构建执行计划（拓扑排序）
    executionPlan, err := e.buildExecutionPlan(flow.CanvasData)
    if err != nil {
        return nil, err
    }
    
    // 2. 按顺序执行节点
    contextData := make(map[string]interface{})  // 存储节点输出
    contextData["start"] = input  // 起始节点输入
    
    for _, node := range executionPlan {
        // 记录节点开始
        nodeLog := &entity.RunNodeLog{
            ID:        uuid.New(),
            RunID:     runID,
            NodeID:    node.ID,
            Status:    "running",
            StartedAt: time.Now(),
        }
        
        if err := e.nodeLogRepo.Create(ctx, nodeLog); err != nil {
            return nil, err
        }
        
        // 执行节点
        output, err := e.executeNode(ctx, node, contextData, runID)
        
        // 更新节点日志
        nodeLog.Status = "success"
        nodeLog.OutputData = output
        nodeLog.DurationMs = int(time.Since(nodeLog.StartedAt).Milliseconds())
        
        if err != nil {
            nodeLog.Status = "failed"
            nodeLog.ErrorMessage = err.Error()
        }
        
        if err := e.nodeLogRepo.Update(ctx, nodeLog); err != nil {
            return nil, err
        }
        
        if err != nil {
            return nil, err
        }
        
        // 保存节点输出到上下文
        contextData[node.ID] = output
    }
    
    // 3. 返回最终输出（end节点的输出）
    return contextData["end"].(map[string]interface{}), nil
}
```

#### 12.2.6 执行单个节点
```go
func (e *FlowExecutor) executeNode(
    ctx context.Context,
    node *Node,
    contextData map[string]interface{},
    runID uuid.UUID,
) (map[string]interface{}, error) {
    switch node.Type {
    case "start":
        return contextData["start"].(map[string]interface{}), nil
        
    case "end":
        // 从上游节点获取输入
        upstreamData := e.getUpstreamData(node, contextData)
        return upstreamData, nil
        
    case "tool":
        return e.executeToolNode(ctx, node, contextData, runID)
        
    default:
        return nil, fmt.Errorf("unknown node type: %s", node.Type)
    }
}

func (e *FlowExecutor) executeToolNode(
    ctx context.Context,
    node *Node,
    contextData map[string]interface{},
    runID uuid.UUID,
) (map[string]interface{}, error) {
    // 1. 查询工具
    toolID := node.Data.ToolID
    tool, err := e.toolRepo.FindByID(ctx, toolID)
    if err != nil {
        return nil, err
    }
    
    // 2. 查询资源
    resource, err := e.resourceRepo.FindByID(ctx, tool.ResourceID)
    if err != nil {
        return nil, err
    }
    
    // 3. 获取插件
    plugin, err := e.pluginRegistry.GetResource(resource.Type)
    if err != nil {
        return nil, err
    }
    
    // 4. 参数映射（将上游输出映射到工具输入）
    input := e.mapInput(node, contextData)
    
    // 5. 执行工具
    result, err := plugin.ExecuteTool(ctx, resource.Config, tool.Config, input)
    if err != nil {
        return nil, err
    }
    
    if result.StatusCode >= 400 {
        return nil, fmt.Errorf("tool execution failed: %s", result.Error)
    }
    
    return result.Data.(map[string]interface{}), nil
}
```

### 12.3 发布流程
```go
func (s *FlowService) PublishFlow(ctx context.Context, id string) (*entity.Flow, error) {
    flow, err := s.flowRepo.FindByID(ctx, uuid.MustParse(id))
    if err != nil {
        return nil, err
    }
    
    // 验证流程完整性
    if err := s.validateCanvasData(flow.CanvasData); err != nil {
        return nil, err
    }
    
    // 发布（状态变更 + 版本号+1）
    flow.Status = "published"
    flow.CurrentVersion++
    flow.UpdatedAt = time.Now()
    
    if err := s.flowRepo.Update(ctx, flow); err != nil {
        return nil, err
    }
    
    return flow, nil
}
```

```go
func (h *FlowHandler) CreateFlow(c *gin.Context) {
    var req dto.CreateFlowRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        response.BadRequest(c, err.Error())
```
