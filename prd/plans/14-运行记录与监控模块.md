#### 14.2.1 查询运行记录列表
```go
func (s *RunService) ListRuns(ctx context.Context, req *dto.RunListRequest) (*dto.RunListResponse, error) {
    filters := map[string]interface{}{
        "triggerId": req.TriggerID,
        "flowId":    req.FlowID,
        "status":    req.Status,
        "startDate": req.StartDate,
        "endDate":   req.EndDate,
    }
    
    runs, total, err := s.runRepo.ListWithPagination(
        ctx,
        uuid.MustParse(req.WorkspaceID),
        filters,
        req.Page,
        req.PageSize,
    )
    if err != nil {
        return nil, err
    }
    
    // 转换为响应格式
    var summaries []*dto.RunSummary
    for _, run := range runs {
        // 查询关联的触发器和流程名称
        trigger, _ := s.triggerRepo.FindByID(ctx, run.TriggerID)
        flow, _ := s.flowRepo.FindByID(ctx, run.FlowID)
        
        summary := &dto.RunSummary{
            ID:           run.ID.String(),
            TriggerID:    run.TriggerID.String(),
            TriggerName:  trigger.Name,
            FlowID:       run.FlowID.String(),
            FlowName:     flow.Name,
            Status:       run.Status,
            InputData:    run.InputData,
            OutputData:   run.OutputData,
            ErrorMessage: run.ErrorMessage,
            StartedAt:    run.StartedAt,
            FinishedAt:   run.FinishedAt,
            DurationMs:   run.DurationMs,
        }
        summaries = append(summaries, summary)
    }
    
    return &dto.RunListResponse{
        Total: total,
        List:  summaries,
    }, nil
}
```

#### 14.2.2 查询运行详情
```go
func (s *RunService) GetRunDetail(ctx context.Context, id string) (*dto.RunDetailResponse, error) {
    // 1. 查询运行记录
    run, err := s.runRepo.FindByID(ctx, uuid.MustParse(id))
    if err != nil {
        return nil, err
    }
    
    // 2. 查询关联信息
    trigger, _ := s.triggerRepo.FindByID(ctx, run.TriggerID)
    flow, _ := s.flowRepo.FindByID(ctx, run.FlowID)
    
    // 3. 查询节点日志
    nodeLogs, err := s.nodeLogRepo.ListByRun(ctx, run.ID)
    if err != nil {
        return nil, err
    }
    
    // 4. 转换为详细响应
    var logDetails []dto.NodeLogDetail
    for _, log := range nodeLogs {
        tool, _ := s.toolRepo.FindByID(ctx, log.ToolID)
        
        detail := dto.NodeLogDetail{
            ID:         log.ID.String(),
            NodeID:     log.NodeID,
            ToolID:     log.ToolID.String(),
            ToolName:   tool.Name,
            ToolMethod: tool.Method,
            Status:     log.Status,
            InputData:  log.InputData,
            OutputData: log.OutputData,
            ErrorMessage: log.ErrorMessage,
            StartedAt:  log.StartedAt,
            DurationMs: log.DurationMs,
        }
        logDetails = append(logDetails, detail)
    }
    
    return &dto.RunDetailResponse{
        RunSummary: dto.RunSummary{
            ID:           run.ID.String(),
            TriggerID:    run.TriggerID.String(),
            TriggerName:  trigger.Name,
            FlowID:       run.FlowID.String(),
            FlowName:     flow.Name,
            Status:       run.Status,
            InputData:    run.InputData,
            OutputData:   run.OutputData,
            ErrorMessage: run.ErrorMessage,
            StartedAt:    run.StartedAt,
            FinishedAt:   run.FinishedAt,
            DurationMs:   run.DurationMs,
        },
        NodeLogs: logDetails,
    }, nil
}
```

#### 14.2.3 统计查询
```go
func (s *RunService) GetRunStats(ctx context.Context, workspaceID string) (map[string]interface{}, error) {
    // 今日统计
    todayStart := time.Now().Truncate(24 * time.Hour)
    
    stats := map[string]interface{}{
        "todayTotal":  s.runRepo.CountByDate(ctx, uuid.MustParse(workspaceID), todayStart, time.Now()),
        "todaySuccess": s.runRepo.CountByStatusAndDate(ctx, uuid.MustParse(workspaceID), "success", todayStart, time.Now()),
        "todayFailed":  s.runRepo.CountByStatusAndDate(ctx, uuid.MustParse(workspaceID), "failed", todayStart, time.Now()),
        "avgDuration":  s.runRepo.AvgDurationByDate(ctx, uuid.MustParse(workspaceID), todayStart, time.Now()),
    }
    
    return stats, nil
}
```

### 14.4 执行时间线视图
```go
type ExecutionTimeline struct {
    RunID       string    `json:"runId"`
    Timeline    []TimelineItem `json:"timeline"`
}

type TimelineItem struct {
    Timestamp time.Time `json:"timestamp"`
    Event     string    `json:"event"`
    NodeID    string    `json:"nodeId,omitempty"`
    Status    string    `json:"status,omitempty"`
    Duration  int       `json:"duration,omitempty"`
}

func buildExecutionTimeline(run *entity.Run, nodeLogs []*entity.RunNodeLog) *ExecutionTimeline {
    // 构建时间线：
    // 1. 流程启动
    // 2. 各节点开始/结束
    // 3. 流程完成
    
    var items []TimelineItem
    
    // 流程启动
    items = append(items, TimelineItem{
        Timestamp: run.StartedAt,
        Event:     "flow_started",
    })
    
    // 节点执行
    for _, log := range nodeLogs {
        items = append(items, TimelineItem{
            Timestamp: log.StartedAt,
            Event:     "node_started",
            NodeID:    log.NodeID,
        })
        
        if log.Status != "running" {
            finishedAt := log.StartedAt.Add(time.Duration(log.DurationMs) * time.Millisecond)
            items = append(items, TimelineItem{
                Timestamp: finishedAt,
                Event:     "node_" + log.Status,
                NodeID:    log.NodeID,
                Status:    log.Status,
                Duration:  log.DurationMs,
            })
        }
    }
    
    // 流程完成
    if run.FinishedAt != nil {
        items = append(items, TimelineItem{
            Timestamp: *run.FinishedAt,
            Event:     "flow_" + run.Status,
            Status:    run.Status,
            Duration:  run.DurationMs,
        })
    }
    
    return &ExecutionTimeline{
        RunID:    run.ID.String(),
        Timeline: items,
    }
}
```
