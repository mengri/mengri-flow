#### 15.1.2 环境服务实现

type EnvironmentService struct {
    envRepo repository.EnvironmentRepository
}

func (s *EnvironmentService) CreateEnvironment(ctx context.Context, req *dto.CreateEnvironmentRequest) (*entity.Environment, error) {
    // 验证Key唯一性
    if existing, _ := s.envRepo.FindByKey(ctx, req.Key); existing != nil {
        return nil, fmt.Errorf("environment key already exists: %s", req.Key)
    }
    
    env := &entity.Environment{
        ID:          uuid.New(),
        Name:        req.Name,
        Key:         req.Key,
        Description: req.Description,
        Color:       req.Color,
        CreatedAt:   time.Now(),
        UpdatedAt:   time.Now(),
    }
    
    if err := s.envRepo.Create(ctx, env); err != nil {
        return nil, err
    }
    
    return env, nil
}

func (s *EnvironmentService) ListEnvironments(ctx context.Context) ([]*entity.Environment, error) {
    return s.envRepo.List(ctx)
}

func (s *EnvironmentService) DeleteEnvironment(ctx context.Context, id string) error {
    // 检查是否有集群
    clusters, err := s.clusterRepo.ListByEnvironment(ctx, uuid.MustParse(id))
    if err != nil {
        return err
    }
    
    if len(clusters) > 0 {
        return fmt.Errorf("cannot delete environment with %d clusters", len(clusters))
    }
    
    return s.envRepo.Delete(ctx, uuid.MustParse(id))
}
```

type ClusterService struct {
    clusterRepo repository.ClusterRepository
    etcdClient  *clientv3.Client
}

func (s *ClusterService) CreateCluster(ctx context.Context, req *dto.CreateClusterRequest) (*entity.Cluster, error) {
    // 验证etcd连接
    if err := s.testEtcdConnection(req.EtcdEndpoints, req.EtcdUsername, req.EtcdPassword); err != nil {
        return nil, fmt.Errorf("etcd connection failed: %w", err)
    }
    
    cluster := &entity.Cluster{
        ID:            uuid.New(),
        Name:          req.Name,
        EnvironmentID: uuid.MustParse(req.EnvironmentID),
        EtcdEndpoints: req.EtcdEndpoints,
        EtcdUsername:  req.EtcdUsername,
        EtcdPassword:  req.EtcdPassword,
        EtcdPrefix:    fmt.Sprintf("/clusters/%s", uuid.New()),
        Description:   req.Description,
        Status:        "active",
        CreatedAt:     time.Now(),
        UpdatedAt:     time.Now(),
    }
    
    if err := s.clusterRepo.Create(ctx, cluster); err != nil {
        return nil, err
    }
    
    return cluster, nil
}

func (s *ClusterService) testEtcdConnection(endpoints []string, username, password string) error {
    cfg := clientv3.Config{
        Endpoints:   endpoints,
        Username:    username,
        Password:    password,
        DialTimeout: 5 * time.Second,
    }
    
    client, err := clientv3.New(cfg)
    if err != nil {
        return err
    }
    defer client.Close()
    
    ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
    defer cancel()
    
    _, err = client.Get(ctx, "/")
    return err
}
```

### 15.3 部署面板

#### 15.3.1 查询集群详情（含执行器列表）
```go
func (s *ClusterService) GetClusterDetail(ctx context.Context, id string) (*dto.ClusterDetailResponse, error) {
    cluster, err := s.clusterRepo.FindByID(ctx, uuid.MustParse(id))
    if err != nil {
        return nil, err
    }
    
    // 从etcd查询执行器状态
    executors, err := s.listExecutorsFromEtcd(ctx, cluster)
    if err != nil {
        return nil, err
    }
    
    // 转换为响应
    response := &dto.ClusterDetailResponse{
        ClusterResponse: dto.ClusterResponse{
            // ... 填充集群基本信息
        },
        Executors: executors,
    }
    
    return response, nil
}

func (s *ClusterService) listExecutorsFromEtcd(ctx context.Context, cluster *entity.Cluster) ([]dto.ExecutorInfo, error) {
    // 查询etcd: /clusters/{cluster-id}/executors/
    prefix := fmt.Sprintf("%s/executors/", cluster.EtcdPrefix)
    
    resp, err := s.etcdClient.Get(ctx, prefix, clientv3.WithPrefix())
    if err != nil {
        return nil, err
    }
    
    var executors []dto.ExecutorInfo
    
    for _, kv := range resp.Kvs {
        var info dto.ExecutorInfo
        if err := json.Unmarshal(kv.Value, &info); err != nil {
            continue
        }
        executors = append(executors, info)
    }
    
    return executors, nil
}
```
