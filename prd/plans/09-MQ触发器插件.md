#### 9.1.4 Start方法实现
```go
func (p *RabbitMQTriggerPlugin) Start(
    ctx context.Context,
    config map[string]interface{},
    handler plugin.TriggerHandler,
) error {
    p.mu.Lock()
    defer p.mu.Unlock()
    
    triggerID := config["triggerId"].(string)
    
    // 初始化映射
    if p.channels == nil {
        p.channels = make(map[string]*amqp.Channel)
        p.configs = make(map[string]map[string]interface{})
        p.handlers = make(map[string]plugin.TriggerHandler)
    }
    
    p.configs[triggerID] = config
    p.handlers[triggerID] = handler
    
    // 连接RabbitMQ
    brokerUrl := config["brokerUrl"].(string)
    conn, err := amqp.Dial(brokerUrl)
    if err != nil {
        return fmt.Errorf("failed to connect to RabbitMQ: %w", err)
    }
    p.conn = conn
    
    // 创建channel
    ch, err := conn.Channel()
    if err != nil {
        return fmt.Errorf("failed to create channel: %w", err)
    }
    
    // 设置QoS
    prefetchCount := 10
    if val, ok := config["prefetchCount"].(float64); ok {
        prefetchCount = int(val)
    }
    ch.Qos(prefetchCount, 0, false)
    
    queue := config["queue"].(string)
    consumerTag := config["consumerTag"].(string)
    autoAck := false
    if val, ok := config["autoAck"].(bool); ok {
        autoAck = val
    }
    
    // 消费消息
    msgs, err := ch.Consume(
        queue,        // queue
        consumerTag,  // consumer
        autoAck,      // auto-ack
        false,        // exclusive
        false,        // no-local
        false,        // no-wait
        nil,          // args
    )
    if err != nil {
        return fmt.Errorf("failed to register consumer: %w", err)
    }
    
    p.channels[triggerID] = ch
    
    // 在后台goroutine中处理消息
    go func() {
        for msg := range msgs {
            p.handleMessage(ctx, triggerID, msg, autoAck)
        }
    }()
    
    return nil
}
```

#### 9.1.5 消息处理
```go
func (p *RabbitMQTriggerPlugin) handleMessage(
    ctx context.Context,
    triggerID string,
    msg amqp.Delivery,
    autoAck bool,
) {
    config := p.configs[triggerID]
    handler := p.handlers[triggerID]
    
    // 解析消息体
    var body interface{}
    if err := json.Unmarshal(msg.Body, &body); err != nil {
        body = string(msg.Body)
    }
    
    // 构造输入数据
    input := map[string]interface{}{
        "messageId": msg.MessageId,
        "body":      body,
        "headers":   msg.Headers,
        "timestamp": time.Now().Format(time.RFC3339),
    }
    
    // 调用handler
    result, err := handler(ctx, input)
    
    // 确认消息
    if !autoAck {
        if err != nil || !result.Success {
            msg.Nack(false, true)  // 重新入队
        } else {
            msg.Ack(false)
        }
    }
}
```

#### 9.1.6 Stop方法
```go
func (p *RabbitMQTriggerPlugin) Stop() error {
    p.mu.Lock()
    defer p.mu.Unlock()
    
    // 关闭所有channel
    for triggerID, ch := range p.channels {
        ch.Close()
        fmt.Printf("RabbitMQ channel for trigger %s closed\n", triggerID)
    }
    
    // 关闭连接
    if p.conn != nil {
        p.conn.Close()
    }
    
    // 清空映射
    p.channels = nil
    p.configs = nil
    p.handlers = nil
    
    return nil
}
```

#### 9.2.2 Kafka消费者实现
```go
func (p *KafkaTriggerPlugin) Start(/* ... */) error {
    // 创建kafka.Reader
    brokers := config["brokers"].([]string)
    topic := config["topic"].(string)
    groupId := config["groupId"].(string)
    
    reader := kafka.NewReader(kafka.ReaderConfig{
        Brokers:   brokers,
        Topic:     topic,
        GroupID:   groupId,
        MinBytes:  10e3, // 10KB
        MaxBytes:  10e6, // 10MB
    })
    
    // 在后台消费消息
    go func() {
        for {
            msg, err := reader.ReadMessage(ctx)
            if err != nil {
                break
            }
            p.handleKafkaMessage(ctx, triggerID, msg)
        }
    }()
    
    return nil
}
```
