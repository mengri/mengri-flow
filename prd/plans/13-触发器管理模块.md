#### 13.2.1 创建触发器
```go
func (s *TriggerService) CreateTrigger(ctx context.Context, req *dto.CreateTriggerRequest) (*entity.Trigger, error) {
    // 1. 验证流程存在且已发布
    flow, err := s.flowRepo.FindByID(ctx, uuid.MustParse(req.FlowID))
    if err != nil {
        return nil, err
    }
    
    if flow.Status != "published" {
        return nil, fmt.Errorf("flow must be published")
    }
    
    // 2. 验证集群存在且正常
    cluster, err := s.clusterRepo.FindByID(ctx, uuid.MustParse(req.ClusterID))
    if err != nil {
        return nil, err
    }
    
    if cluster.Status != "active" {
        return nil, fmt.Errorf("cluster is not active")
    }
    
    // 3. 验证触发器插件
    triggerPlugin, err := s.pluginRegistry.GetTrigger(req.Type)
    if err != nil {
        return nil, fmt.Errorf("unsupported trigger type: %s", req.Type)
    }
    
    // 4. 验证配置Schema
    schema := triggerPlugin.ConfigSchema()
    if err := validateSchema(schema, req.Config); err != nil {
        return nil, fmt.Errorf("invalid config: %w", err)
    }
    
    // 5. 创建触发器实体
    trigger := &entity.Trigger{
        ID:            uuid.New(),
        Name:          req.Name,
        Type:          req.Type,
        Config:        req.Config,
        FlowID:        uuid.MustParse(req.FlowID),
        FlowVersion:   req.FlowVersion,
        ClusterID:     uuid.MustParse(req.ClusterID),
        InputMapping:  req.InputMapping,
        OutputMapping: req.OutputMapping,
        ErrorHandling: req.ErrorHandling,
        WorkspaceID:   uuid.MustParse(req.WorkspaceID),
        Status:        "paused",  // 默认停用，需手动启用
        CreatedAt:     time.Now(),
        UpdatedAt:     time.Now(),
    }
    
    if err := s.triggerRepo.Create(ctx, trigger); err != nil {
        return nil, err
    }
    
    return trigger, nil
}
```

#### 13.2.2 启用/停用触发器
```go
func (s *TriggerService) EnableTrigger(ctx context.Context, id string) error {
    trigger, err := s.triggerRepo.FindByID(ctx, uuid.MustParse(id))
    if err != nil {
        return err
    }
    
    // 验证依赖完整性
    if err := s.validateTriggerDependencies(ctx, trigger); err != nil {
        return fmt.Errorf("cannot enable trigger: %w", err)
    }
    
    trigger.Status = "active"
    trigger.UpdatedAt = time.Now()
    
    if err := s.triggerRepo.Update(ctx, trigger); err != nil {
        return err
    }
    
    return nil
}

func (s *TriggerService) DisableTrigger(ctx context.Context, id string) error {
    trigger, err := s.triggerRepo.FindByID(ctx, uuid.MustParse(id))
    if err != nil {
        return err
    }
    
    trigger.Status = "paused"
    trigger.UpdatedAt = time.Now()
    
    return s.triggerRepo.Update(ctx, trigger)
}
```

#### 13.2.3 发布到集群
```go
func (s *TriggerService) PublishToCluster(ctx context.Context, triggerID string, clusterID string) error {
    // 1. 查询触发器
    trigger, err := s.triggerRepo.FindByID(ctx, uuid.MustParse(triggerID))
    if err != nil {
        return err
    }
    
    // 2. 查询集群
    cluster, err := s.clusterRepo.FindByID(ctx, uuid.MustParse(clusterID))
    if err != nil {
        return err
    }
    
    // 3. 查询流程
    flow, err := s.flowRepo.FindByID(ctx, trigger.FlowID)
    if err != nil {
        return err
    }
    
    // 4. 查询流程引用的所有工具和资源
    tools, err := s.collectFlowTools(ctx, flow)
    if err != nil {
        return err
    }
    
    resources, err := s.collectFlowResources(ctx, tools)
    if err != nil {
        return err
    }
    
    // 5. 写入etcd
    etcdKeyPrefix := fmt.Sprintf("/clusters/%s", cluster.ID)
    
    // 写入触发器配置
    triggerKey := fmt.Sprintf("%s/triggers/%s", etcdKeyPrefix, trigger.ID)
    triggerData, _ := json.Marshal(trigger)
    if _, err := s.etcdClient.Put(ctx, triggerKey, string(triggerData)); err != nil {
        return err
    }
    
    // 写入流程定义
    flowKey := fmt.Sprintf("%s/flows/%s", etcdKeyPrefix, flow.ID)
    flowData, _ := json.Marshal(flow)
    if _, err := s.etcdClient.Put(ctx, flowKey, string(flowData)); err != nil {
        return err
    }
    
    // 写入工具定义
    for _, tool := range tools {
        toolKey := fmt.Sprintf("%s/tools/%s", etcdKeyPrefix, tool.ID)
        toolData, _ := json.Marshal(tool)
        if _, err := s.etcdClient.Put(ctx, toolKey, string(toolData)); err != nil {
            return err
        }
    }
    
    // 写入资源配置
    for _, resource := range resources {
        resourceKey := fmt.Sprintf("%s/resources/%s", etcdKeyPrefix, resource.ID)
        resourceData, _ := json.Marshal(resource)
        if _, err := s.etcdClient.Put(ctx, resourceKey, string(resourceData)); err != nil {
            return err
        }
    }
    
    return nil
}
```

#### 13.2.4 删除触发器

```go
func (s *TriggerService) DeleteTrigger(ctx context.Context, id string) error {
    // 1. 检查状态
    trigger, err := s.triggerRepo.FindByID(ctx, uuid.MustParse(id))
    if err != nil {
        return err
    }
    
    if trigger.Status == "active" {
        return fmt.Errorf("cannot delete active trigger, disable first")
    }
    
    // 2. 删除
    return s.triggerRepo.Delete(ctx, uuid.MustParse(id))
}
```

### 13.4 输入输出映射
```go
func applyInputMapping(input map[string]interface{}, mapping map[string]interface{}) map[string]interface{} {
    result := make(map[string]interface{})
    
    for source, target := range mapping {
        // source: "request.body.userId"
        // target: "userId"
        
        value := getValueByPath(input, source)
        setValueByPath(result, target.(string), value)
    }
    
    return result
}

func getValueByPath(data map[string]interface{}, path string) interface{} {
    // 实现：根据路径提取值，如 "request.body.userId"
    parts := strings.Split(path, ".")
    current := data
    
    for i, part := range parts {
        if i == len(parts)-1 {
            return current[part]
        }
        
        next, ok := current[part].(map[string]interface{})
        if !ok {
            return nil
        }
        current = next
    }
    
    return nil
}
```
